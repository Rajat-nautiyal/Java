GFG:- Bellman-Ford

//TC:- O(V*E) SC:- O(V);
class Solution {
    public int[] bellmanFord(int V, int[][] edges, int src) {
        // code here
        int[] dist = new int[V];
        Arrays.fill(dist, (int)1e8);
        dist[src] = 0;
        
        for(int i = 0; i < V; i++){
            boolean updated = false;
            for(int[] e : edges){
                int u = e[0];
                int v = e[1];
                int wt = e[2];
                if(dist[u] != 1e8 && dist[u] + wt < dist[v]){
                    dist[v] = dist[u] + wt;
                    updated = true;
                }
            }
            if(!updated) break; //early termination for good TC
        }
        //detect -ve edge loop
        for(int[] e : edges){
            int u = e[0];
            int v = e[1];
            int wt = e[2];
            if(dist[u] != 1e8 && dist[u] + wt < dist[v]){
                return new int[]{-1};
            }
        }
        return dist;
    }
}
