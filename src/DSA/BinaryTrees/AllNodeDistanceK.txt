863. All Nodes Distance K in Binary Tree

//BFS APPROACH and there is also a dfs approach
class Solution {
    public void trackParent(TreeNode root, HashMap<TreeNode, TreeNode> parent_track){
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);        
        while(!q.isEmpty()){
            int size = q.size();
            for(int i = 0; i < size; i++){
                TreeNode parent = q.poll();
                if(parent.left != null){
                    parent_track.put(parent.left, parent);
                    q.offer(parent.left);
                }
                if(parent.right != null){
                    parent_track.put(parent.right, parent);
                    q.offer(parent.right);
                }
            }
        }
    }
    public List<Integer> distanceK(TreeNode root, TreeNode target, int k) {
        HashMap<TreeNode, TreeNode> parent_track = new HashMap<>();
        trackParent(root, parent_track);
        HashMap<TreeNode, Boolean> visited = new HashMap<>();
        Queue<TreeNode> q = new LinkedList<>();
        List<Integer> ans = new ArrayList<>();
        q.offer(target);
        visited.put(target, true);
        int currsize = 0;
        while(!q.isEmpty()){
            int size = q.size();
            if(currsize == k) break;
            currsize++;
            for(int i = 0; i < size; i++){
                TreeNode node = q.poll();
                if(node.left != null && visited.get(node.left) == null){
                    q.offer(node.left);
                    visited.put(node.left, true);
                }
                if(node.right != null && visited.get(node.right) == null){
                    q.offer(node.right);
                    visited.put(node.right, true);
                }
                TreeNode parent = parent_track.get(node);
                if(parent != null && visited.get(parent) == null){
                    q.offer(parent);
                    visited.put(parent, true);
                }
            }
        }
        while(!q.isEmpty()){
            ans.add(q.poll().val);
        }
        return ans;
    }
}
