class Solution {
    public int rec(TreeNode node, int[] max){
        if(node == null) return 0;

        int lv = rec(node.left, max);
        int rv = rec(node.right, max);
        if(lv < 0) lv = 0;
        if(rv < 0) rv = 0;
        max[0] = Math.max(max[0], lv+rv+node.val);
        return Math.max(lv,rv)+node.val;
    }
    public int maxPathSum(TreeNode root) {
        int min = Integer.MIN_VALUE;
        int[] max = {min};
        rec(root, max);
        return max[0];
    }
}
