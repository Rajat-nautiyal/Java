class Solution {
    public boolean isLeaf(Node node){
        return (node.left == null) && (node.right == null);
    }
    public void addLeftBoundary(Node node, ArrayList<Integer> ans){
        Node curr = node.left;
        while(curr != null){
            if(!isLeaf(curr)) ans.add(curr.data);
            if(curr.left != null) curr = curr.left;
            else curr = curr.right;
        }
    }
    public void addRightBoundary(Node node, ArrayList<Integer> ans){
        Node curr = node.right;
        List<Integer> temp = new ArrayList<>();
        while(curr != null){
            if(!isLeaf(curr)) temp.add(curr.data);
            if(curr.right != null) curr = curr.right;
            else curr = curr.left;
        }
        for(int i = temp.size() - 1; i >= 0; i--){
            ans.add(temp.get(i));
        }
    }
    public void addLeafNodes(Node node, ArrayList<Integer> ans){
        if(isLeaf(node)){
            ans.add(node.data);
        }
        if(node.left != null) addLeafNodes(node.left, ans);
        if(node.right != null) addLeafNodes(node.right, ans);
        
    }
    //Main Function
    ArrayList<Integer> boundaryTraversal(Node root) {
        ArrayList<Integer> ans = new ArrayList<>();
        if(root == null) return ans;
        
        if(!isLeaf(root)){
            ans.add(root.data);
        }
        //Iterate left boundary
        addLeftBoundary(root, ans);
        
        //Add leaf nodes(left to right)
        addLeafNodes(root, ans);
        
        //Iterate Right boundary
        addRightBoundary(root, ans);
        
        return ans;
    }
}
