Q.  Children Sum Property

public class Solution {
    public static void helper(BinaryTreeNode <Integer> node){
        if(node == null) return;
        if(node.left != null && node.data > node.left.data){
            node.left.data = node.data;
        }
        if(node.right != null && node.data > node.right.data){
            node.right.data = node.data;
        }
        helper(node.left);
        helper(node.right);

        if(node.left != null && node.right != null) node.data = node.left.data + node.right.data;
        else if(node.left != null) node.data = node.left.data;
        else if(node.right != null) node.data = node.right.data;

    }
    public static void changeTree(BinaryTreeNode < Integer > root) {
        helper(root);
    }
}
