257. Binary Tree Paths

class Solution {
    public void helper(TreeNode node, List<String> ans, String s){
        if(node == null) return;
        if(node.left == null && node.right == null){
            ans.add(s + node.val);         
        } 
        helper(node.left, ans, s + node.val + "->");
        helper(node.right, ans, s + node.val + "->");
    }
    public List<String> binaryTreePaths(TreeNode root) {
        List<String> ans = new ArrayList<>();
        helper(root, ans, "");
        return ans;
    }
}

    public void helper(Node node, ArrayList<ArrayList<Integer>> ans, ArrayList<Integer> temp){
        if(node == null) return;
        temp.add(node.data);
        if(node.left == null && node.right == null){
            ans.add(new ArrayList<>(temp));
        }
        helper(node.left, ans, temp);
        helper(node.right, ans, temp);
        temp.remove(temp.size()-1);
        
    }
    public ArrayList<ArrayList<Integer>> Paths(Node root) {
        // code here
        ArrayList<ArrayList<Integer>> ans = new ArrayList<>();
        ArrayList<Integer> temp = new ArrayList<>();
        helper(root, ans, temp);
        return ans;
    }
