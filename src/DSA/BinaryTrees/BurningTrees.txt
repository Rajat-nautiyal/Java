GFG Q. Burning Tree(SIMILAR LC 2385)

class Solution {
    public Node findTarget(Node node, int target){
        if(node == null) return null;
        if(node.data == target) return node;
        
        Node l = findTarget(node.left, target);
        Node r = findTarget(node.right, target);
        
        if(l != null && l.data == target) return l;
        else return r;
    }
    public void parentTracing(Node root, HashMap<Node, Node> parent_track){
        Queue<Node> q = new LinkedList<>();
        q.offer(root);
        while(!q.isEmpty()){
            int n = q.size();
            for(int i = 0; i < n; i++){
                Node node = q.poll();
                if(node.left != null){
                    parent_track.put(node.left, node);
                    q.offer(node.left);
                }
                if(node.right != null){
                    parent_track.put(node.right, node);
                    q.offer(node.right);
                }
            }
        }
    }
    public int minTime(Node root, int target) {
        // code here
        HashMap<Node, Node> parent_track = new HashMap<>();
        HashMap<Node, Boolean> visited = new HashMap<>();
        Node tar = findTarget(root, target);
        parentTracing(root, parent_track);
        Queue<Node> q = new LinkedList<>();
        q.offer(tar);
        visited.put(tar, true);
        int ans = -1;
        while(!q.isEmpty()){
            int n = q.size();
            ans++;
            for(int i = 0; i < n; i++){
                Node node = q.poll();
                if(node.left != null && visited.get(node.left) == null){
                    q.offer(node.left);
                    visited.put(node.left, true);
                }
                if(node.right != null && visited.get(node.right) == null){
                    q.offer(node.right);
                    visited.put(node.right, true);
                }
                Node parent = parent_track.get(node);
                if(parent != null && visited.get(parent) == null){
                    q.offer(parent);
                    visited.put(parent, true);
                }

            }
        }
        return ans;
    }
}

//DFS approach - just get the max distance from target node
    public int dfs(Node node, HashMap<Node, Boolean> visited, HashMap<Node, Node> parent_track ){
        if(node == null) return -1;
        if(visited.containsKey(node)) return -1;
        visited.put(node, true);
        Node parent = parent_track.get(node);
        int p = dfs(parent, visited, parent_track);
        int l = dfs(node.left, visited, parent_track);
        int r = dfs(node.right, visited, parent_track);
        int max = Math.max(l,r);
        max = Math.max(max,p);
        return max+1;
    }
    public int minTime(Node root, int target){
        // code here
        HashMap<Node, Node> parent_track = new HashMap<>();
        parentTracing(root, parent_track);
        Node tar = findTarget(root, target);
        return dfs(tar, new HashMap<>() , parent_track);
    }
