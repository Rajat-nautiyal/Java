//Ceil in BST

//Iterative Approach(Optimised)  TC - O(logN) SC - O(1)
class Solution {
    int findCeil(Node root, int x) {
        // code here
        int ceil = -1;
        
        while(root != null){
            if(root.data == x) return x;
            
            if(root.data > x){
                ceil = root.data;
                root = root.left;
            }else root = root.right;
        }
        return ceil;
        
    }
}

//recursion approach  TC - O(logN) SC - O(logN)
class Solution {
    void helper(Node node, int x, int[] ans){
        if(node == null) return;
        
        if(node.data == x){
          ans[0] = node.data;
          return;
        } 
        if(node.data > x){
            ans[0] = node.data;
            helper(node.left, x, ans);
        } 
        else helper(node.right, x, ans);
    }
    int findCeil(Node root, int x) {
        int[] ans = {-1};
        helper(root, x, ans);
        return ans[0];
    }
}

//Brute Approach:- level order traversal approach TC- O(N) SC-O(N)
class Solution {
    int findCeil(Node root, int x) {
        // code here
        int ans = Integer.MAX_VALUE;
        Queue<Node> q = new LinkedList<>();
        q.offer(root);
        if(root.data == x) return root.data;
        else if(root.data > x) ans = root.data;
        
        while(!q.isEmpty()){
            int n = q.size();
            
            for(int i = 0; i < n; i++){
                Node node = q.poll();
                if(node.left != null) q.offer(node.left);
                if(node.right != null) q.offer(node.right);
                
                if(node.data == x) return x;
                if(node.data > x && ans > node.data) ans = node.data;
            }
        }
        if(ans == Integer.MAX_VALUE) return -1;
        return ans;
    }
}
