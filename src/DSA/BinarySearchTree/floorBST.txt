Floor in a Binary Search Tree

//Iterative Approach(Optimised) TC - O(logN) SC - O(1)
class Solution {
    public int findMaxFork(Node root, int k) {
        // code here.
        int floor = -1;
        
        while(root != null){
            if(root.data == k) return k;
            
            if(root.data < k){
                floor = root.data;
                root = root.right;
            }else root = root.left;
        }
        return floor;
    }
}

//Recursion Approach TC - O(logN) SC - O(logn)
class Solution {
    public void helper(Node node, int k, int[] ans){
        if(node == null) return;
        
        if(node.data == k){
            ans[0] = k;
            return;
        }
        if(node.data < k){
            ans[0] = node.data;
            helper(node.right, k, ans);
        }else helper(node.left, k, ans);
    }
    public int findMaxFork(Node root, int k) {
        // code here.
        int[] ans = {-1};
        helper(root, k, ans);
        return ans[0];
    }
}

//Brute Approach:- level order traversal approach TC- O(N) SC-O(N)
class Solution {
    public int findMaxFork(Node root, int k) {
        // code here.
        int ans = -1;
        Queue<Node> q = new LinkedList<>();
        q.offer(root);
        if(root.data == k) return root.data;
        else if(root.data < k) ans = root.data;
        
        while(!q.isEmpty()){
            int n = q.size();
            
            for(int i = 0; i < n; i++){
                Node node = q.poll();
                if(node.left != null) q.offer(node.left);
                if(node.right != null) q.offer(node.right);
                
                if(node.data == k) return k;
                if(node.data < k && ans < node.data) ans = node.data;
            }
        }
        return ans;
    }
}
