701. Insert into a Binary Search Tree

//Iterative Approach(Optimised): TC - O(logN) SC-(1)
class Solution {
    public TreeNode insertIntoBST(TreeNode root, int val) {
        if(root == null) return new TreeNode(val);
        TreeNode node = root;
        while(node != null){
            if(node.val < val){
                if(node.right == null){
                    node.right = new TreeNode(val);
                    break;
                }else node = node.right;
            }
            if(node.val > val){
                if(node.left == null){
                    node.left = new TreeNode(val);
                    break;
                }else node = node.left;
            }
        }
        return root;
    }
}

//Recursion Approach TC - O(logN) SC-(logn)
class Solution {
    public void helper(TreeNode node, int val){
        if(node.val < val){
            if(node.right == null){
                TreeNode n = new TreeNode(val);
                node.right = n;
                return;
            }else helper(node.right, val);
        }
        if(node.val > val){
            if(node.left == null){
                TreeNode n = new TreeNode(val);
                node.left = n;
                return;
            }else helper(node.left, val);
        }
    }
    public TreeNode insertIntoBST(TreeNode root, int val) {
        if(root == null) return new TreeNode(val);
        helper(root, val);
        return root;
    }
}
