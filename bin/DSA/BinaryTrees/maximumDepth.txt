class Solution {
    public int rec(TreeNode node, int depth){
        if(node == null) return depth - 1;
        int max = depth;
        max = Math.max(depth, rec(node.left, depth+1));
        max = Math.max(max, rec(node.right, depth+1));
        return max;
    }
    public int maxDepth(TreeNode root) {
        if(root == null) return 0;

        //Recursion Approach
        int max = rec(root, 1);
        return max;


        //Level Order Queue Approach
        int lvl = 0;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);

        while(!q.isEmpty()){
            int n = q.size();
            for(int i = 0; i < n; i++){
                TreeNode node = q.poll();
                if(node.left != null) q.offer(node.left);
                if(node.right != null) q.offer(node.right);                
            }
            lvl++;            
        }
        return lvl;
    }
}
