//Morris Inorder Traversal of a Binary Tree

TC - O(N) SC- O(1)
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List<Integer> ans = new ArrayList<>();        
        if(root == null) return ans;

        TreeNode curr = root;
        while(curr != null){
            if(curr.left == null){
                ans.add(curr.val);
                curr = curr.right;
            }else{
                TreeNode p = curr.left;
                while(p.right != null && p.right != curr){
                    p = p.right;
                }
                if(p.right == null){
                    p.right = curr;
                    curr = curr.left;
                }else{
                    p.right = null;
                    ans.add(curr.val);
                    curr = curr.right;
                }
            }
        }
        return ans;
    }
}


//Morris Preorder Traversal of a Binary Tree

    public List<Integer> preorder(TreeNode root) {

        TreeNode curr = root;

                TreeNode p = curr.left;
                while(p.right != null && p.right != curr){
                    p = p.right;
                }
                if(p.right == null){
                    p.right = curr;
                    ans.add(curr.val);
                    curr = curr.left;
                }else{
                    p.right = null;
                    curr = curr.right;
                }
            }
        }

}
