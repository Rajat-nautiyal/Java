1008. Construct Binary Search Tree from Preorder Traversal

//Brute Approach :- Recursion TC-o(n^2) SC-(n);
class Solution {
    public TreeNode helper(int[] preorder, int start, int end){
        if(start > end) return null;

        TreeNode root = new TreeNode(preorder[start]);
        int lstart = start+1;
        int rstart = start+1;
        for(int i = start+1; i <= end; i++){
            if(preorder[i] < root.val) rstart++;
            else break;
        }
        root.left = helper(preorder, lstart, rstart-1);
        root.right = helper(preorder, rstart, end);
        return root;
    }
    public TreeNode bstFromPreorder(int[] preorder) {
        return helper(preorder,0,preorder.length-1);
    }
}

//Better Approach:- sort preorder -> inorder and construct BST from inorder and preorder
TC-o(nlogn) SC-o(n);

//Optimised Approach (Similar to Validate BT)
class Solution {
    public TreeNode helper(int[] preorder, int ubound, int[] i){
        if(i[0] == preorder.length || preorder[i[0]] > ubound) return null;

        TreeNode root = new TreeNode(preorder[i[0]++]);
        root.left = helper(preorder, root.val, i);
        root.right = helper(preorder, ubound, i);
        return root;
    }
    public TreeNode bstFromPreorder(int[] preorder) {
        int[] i = {0};
        return helper(preorder,Integer.MAX_VALUE, i);
    }
}
